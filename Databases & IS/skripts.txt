create table Klients (
	id int identity(1,1) primary key,
	epasts varchar(30) not null,
	parole char(64) not null,
	vards nvarchar(30) not null,
	uzvards nvarchar(30) not null,
	dzimsanas_diena date,
	telefona_nr numeric(11,0)
)

create table KlientsKarte (
	id int identity(1,1) primary key,
	klients_id int foreign key references Klients not null,
	numurs numeric(16,0) not null,
	CVC numeric(3,0) not null,
	termins_lidz date not null
)

create table Adrese (
	id int identity(1,1) primary key,
	klients_id int foreign key references Klients not null,
	pilseta nvarchar(30) not null,
	iela nvarchar(30) not null,
	majas_nr int not null,
	dzivokla_nr int,
	pasta_indekss nvarchar(7) not null
)

create table Pasutijums (
	id int identity(1,1) primary key,
	klients_id int foreign key references Klients not null,
	adrese_id int foreign key references Adrese not null,
	karte_id int foreign key references KlientsKarte not null,
	datums date not null,
	cena smallmoney not null
)

create table Kategorija (
	id int identity(1,1) primary key,
	nosaukums nvarchar(30) not null,
)

create table Prece (
	id int identity(1,1) primary key,
	kategorija_id int foreign key references Kategorija not null,
	nosaukums nvarchar(30) not null,
	apraksts nvarchar(200) not null,
	cena smallmoney not null,
	atlaides_cena smallmoney
)

create table PasutijumsPrece (
	prece_id int foreign key references Prece not null,
	pasutijums_id int foreign key references Pasutijums not null,
	skaits int not null
)

SELECT epasts FROM Klients
WHERE 
	NOT EXISTS(SELECT * FROM KlientsKarte WHERE Klients.id = KlientsKarte.klients_id) 
	AND
	EXISTS(SELECT * FROM Adrese WHERE Klients.id = Adrese.klients_id)
-- Atlasam klientus, kuri nav pievienojuši savu maksājumu karti mājaslapā un atlasam to epastus, lai varētu tiem nosūtīt ziņu,
-- ja tie izlem pievienot savu karti šīs nedēļas laikā, tie saņem atlaidi savam nākamajam pirkumam.

SELECT *
FROM Prece
WHERE Prece.cena > 500 AND Prece.kategorija_id = 2
ORDER BY Prece.nosaukums
-- Atlasam visas preces kategorijā 'Datori' un pārbaudam vai to cenas ir lielākas par 500.00, kā arī
-- kārtojam šīs preces pēc nosaukuma alfabētiskā secībā, šādi simulējot filtrus, ko kāds lietotājs vārētu izvēlēties mājaslapā.

SELECT Klients.epasts, Klients.vards, Klients.uzvards, Klients.telefona_nr, Adrese.pilseta, KlientsKarte.numurs FROM Klients
JOIN Adrese ON Klients.id = Adrese.klients_id
JOIN KlientsKarte ON Klients.id = KlientsKarte.klients_id
WHERE Klients.telefona_nr IS NOT NULL
-- Apvienojam trīs tabulas un no katras tabulas izvēlamies noteiktas kollonas, papildus pārbaudam vai lietotājs ir pievienojis telefona
-- numuru, iegūto informāciju varētu attēlot lietotāja profilā

SELECT pilseta, pasta_indekss, COUNT (id) lietotaju_skaits
FROM Adrese
GROUP BY pilseta, pasta_indekss
-- No adrešu tabulas atlasam pilsētu un pasta indeksu kombinācijas un aprēķinam lietotāju skaitu šajā lokācijā

SELECT Pasutijums.id, PasutijumsPrece.skaits, Prece.nosaukums, Prece.cena, Prece.atlaides_cena, Pasutijums.datums, Pasutijums.cena AS pasutijuma_kopsumma 
FROM PasutijumsPrece
JOIN Prece ON PasutijumsPrece.prece_id = Prece.id
JOIN Pasutijums ON PasutijumsPrece.pasutijums_id = Pasutijums.id
ORDER BY Pasutijums.datums
-- Apvienojam trīs tabulas un no katras tabulas izvēlamies noteiktas kollonas, kārtojam tās pēc pasūtījuma datuma,
-- šo atlasi varētu izmantot, lai pārskatītu pasūtītās preces pasūtījumos

SELECT id, kategorija_id, nosaukums, cena, RANK () OVER ( PARTITION BY kategorija_id ORDER BY cena DESC) cenas_vieta
FROM Prece
-- Atlasam preces pēc to kategorijas un cenas, šos datus varētu izmantot, lai atlasītu dārgākās preces noteiktā kategorijā
-- rank tiek izmantots, lai redzētu preces vietu atkarībā no citām precēm tajā pašā kategorijā.

SELECT DISTINCT datums, SUM (cena) OVER (PARTITION BY datums) ieguvumi
FROM Pasutijums
ORDER BY ieguvumi DESC
-- Atlasam pasūtījumus pēc to veiktā datuma un aprēķinam ieguvumus tajā dienā no visiem veiktajiem pasūtījumiem

SELECT DISTINCT Klients.id, Klients.epasts, Klients.vards, Klients.uzvards, 
	COUNT(*) OVER (PARTITION BY Pasutijums.klients_id) veiktie_pasutijumi,
	SUM(Pasutijums.cena) OVER (PARTITION BY Pasutijums.klients_id) pasutijumu_ieguvumi
FROM Klients, Pasutijums
WHERE Pasutijums.klients_id = Klients.id
-- Apvienojam klientu un pasūtījumu tabulas, iegūstam lietotāja veikto pasūtījumu skaitu, kā arī šo pasūtījumu kopējo cenu
